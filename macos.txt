^ = ctrl

# Keyboard shortcuts
cmd - h // hide all others
cmd - m // minimise
cmd - n // new window
cmd-shift-. // open hidden file

# Switch application
cmd - tab ~ // back menu
cmd - ~ // next window
cmd - shift - ~ // previous window

# Mission control
ctrl - up // switch workspaces
ctrl - down // switch windows
F11 // show desktop
F12 // show dashboard
^-[12] // switch to desktop
click window + ^-left // move window between spaces

# Screenshots
cmd-shift-4 // select region
cmd-shift-4-space // current window
ctrl - ...  // copy to clipboard instead of desktop

# port
selfupdate  // update ports
upgrade outdated  // upgrade all ports
## selecting default group
select --list python  // list options
select --set python27 // make default
## variants
port info vim
port install vim +python27
