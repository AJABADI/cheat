# sbatch
SBATCH_ACCOUNT
SBATCH_JOB_NAME
SBATCH_QOS

sbatch slurm_submit
  -J jobname
  -D workdir
  -o stdout.log (default: slurm*)
  -e stderr.log (default: stdout)
  --mem mem // memory in MB per node
  --mem-per-cpu mem // minimum memory in MB per CPU
  --mincpus n  // min # cpus per node
  --test-only
  --qos=INTR  // for fast running jobs
