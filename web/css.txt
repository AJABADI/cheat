# syntax
.class { }
  <div class='class'/>
  <div class='class1 class2'/>

    * classes can not inherit from each other
div { }
#id {}  // global id
p,div,#id,.class  // or
div.class // div in class; not .class div!
svg div // div in svg element

border-color: black
border: width style color // 3px solid black
fill: color
fill-opacity: [0;1]
