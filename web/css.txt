# syntax
.class { }
  <div class='class'/>
  <div class='class1 class2'/>
    * inherit from multiple classes
    * classes can not inherit from each other
div { }
#id {}  // global id
p,div,#id,.class  // or
div.class // div in class; not .class div!
svg div // div in svg element
