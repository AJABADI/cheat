# overriding
class Parent(object):
  def __init__(self, a):
    self.a = a

class Child(Parent):

c = Child(1)
  * Inherits constructor of parent!
  * Requires a argument!

## overriding constructor and methods
class Child(Parent):
  def __init__(self, a, **kwargs):
    self.a = a
    super(Child, self).__init__(*kwargs)  // call parent constructor

  def method(self):
    super(Child, self).method() // call parent method


# Decorators
@staticmethod
def static():
  // Class.static() or object.static()
  // no self (class) argument
  // like static C++

@classmethod
def static(self):
  // Class.static() or object.static()
  // has self (class) argument


@property // getter
def value(self):
  return self._value

@value.setter // setter
def value(self, new):
  self._value = new
